14 I2C_writeRegister8bit(device_handle, ADXL345_REG_POWER_CTL, 0x08); // Bit 3 high to start measuring

54void I2C_adxl345_setRate(I2C_dev_handles device_handle, double rate) {
55 uint8_t _b, _s; // Variables to hold register values
56 int v = (int) (rate / 6.25); // Convert the desired rate to a value used by the sensor
57 int r = 0; // Variable to determine the bit position for the rate setting
58
59 // Calculate the rate setting value based on the input rate
60 // This loop determines how many times the value can be shifted right
61 // before it becomes zero, effectively calculating the position of the highest set bit.
62 while (v >>= 1) {
63 r++;
64 }
65
66 // The valid range for r is 0 to 9, corresponding to available data rate settings
67 if (r <= 9) {
68 // Read the current value of the BW_RATE register
69 // The BW_RATE register controls the data rate and power mode of the sensor
70 _b = (uint8_t)(I2C_readRegister16bit(device_handle, ADXL345_REG_BW_RATE)>>8);
71
72 // Calculate the new value for the BW_RATE register
73 // The lower 4 bits of the BW_RATE register (D3-D0) determine the output data rate
74 // We preserve the upper 4 bits (D7-D4) by masking with 0xF0 and OR with the new rate setting
75 _s = (uint8_t) (r + 6) | (_b & 0xF0);
76
77 // Write the new value to the BW_RATE register
78 I2C_writeRegister8bit(device_handle, ADXL345_REG_BW_RATE, _s);
79 }
