34 // Create structure for managing each sensor.
35 sensor_t sensors[] = {
36 {I2C_distance_sens1_dev_handle, I2C_distance_sens_1_mux, distance_sens1, distance_sens1, SENSOR_IDLE},
37 {I2C_distance_sens2_dev_handle, I2C_distance_sens_2_mux, distance_sens2, distance_sens2, SENSOR_IDLE},

49 while (1) {
50
51 //ESP_LOG("I2C_TASK","%u ms",pdTICKS_TO_MS(xTaskGetTickCount()));

54 current_sensor_index = (current_sensor_index + 1) % sensor_count;

57 static uint16_t readValSensor_dist1_prev = 0;

64 case SENSOR_IDLE:
65 I2C_select_multiplexer_channel(current_sensor->mux);
66 vTaskDelay(pdMS_TO_TICKS(5)); // Allow for multiplexer switch
67 current_sensor->state = SENSOR_READING;
68 break;

70 case SENSOR_READING:
71 switch (current_sensor->mux) {
72 case I2C_distance_sens_1_mux:
73 uint16_t readValSensor1;
74 // Read new sensor value
75 readValSensor1 = VL53L0X_readRangeContinuousMillimeters(I2C_distance_sens1_dev_handle);
76 // Apply low-pass filter
77 readValSensor_dist1_final = readValSensor1;
78 //readValSensor_dist1_final = ALPHA_VL53L0X * (double)readValSensor1 + (1 - ALPHA_VL53L0X) * (double)readValSensor_dist1_prev;

82

88 countSendStart_sens1=0;

90 i2c_command.commandValue = 0;
91 i2c_command.sendingSensor = I2C_distance_sens1_dev_handle;
92 xQueueSend(I2C_commandQueue,&i2c_command,pdMS_TO_TICKS(5000));
93 //ESP_LOGI("", "Sent STOP for forward \n");

96 {
97 countSendStop_sens1=0;
98 countSendStart_sens1++;
99 i2c_command.command = I2C_START_MOTOR;
100 i2c_command.commandValue = 0;
101 i2c_command.sendingSensor = I2C_distance_sens1_dev_handle;
102 xQueueSend(I2C_commandQueue,&i2c_command,pdMS_TO_TICKS(5000));
103 //ESP_LOGI("", "Sent START for forward \n");
104 }
105
106 // Log the filtered value
107 //ESP_LOGI("I2C", "Range1: [ %.2lf ]", readValSensor_dist1_final);
108 //vTaskDelay(pdMS_TO_TICKS(50)); // Allow for multiplexer switch
109 break;
110
111 case I2C_distance_sens_2_mux:
112 uint16_t readValSensor2;
113 readValSensor2 = VL53L0X_readRangeContinuousMillimeters(current_sensor->device_handle);
114// readValSensor_dist2_final = ALPHA_VL53L0X * (double)readValSensor2 + (1 - ALPHA_VL53L0X) * (double)readValSensor_dist2_prev; // Update the previous value for the next iteration
115// readValSensor_dist2_prev = readValSensor2;
116 readValSensor_dist2_final = readValSensor2;
117 sendCommandApp(DistSensBw, (double*)&readValSensor_dist2_final, DOUBLE);
118 static int countSendStop_sens2 = 0,countSendStart_sens2 = 0;
119 if(readValSensor_dist2_final < (Threshold_dist * speed_distance_sens_scaling) && countSendStop_sens2 == 0)
120 {
121 countSendStop_sens2++;
122 countSendStart_sens2=0;
123 i2c_command.command = I2C_STOP_MOTOR;
124 i2c_command.commandValue = 0;
125 i2c_command.sendingSensor = I2C_distance_sens2_dev_handle;
126 xQueueSend(I2C_commandQueue,&i2c_command,-1);
127 //ESP_LOGI("", "Sent STOP for backward \n");
128 }
129 else if (countSendStart_sens2 == 0 && readValSensor_dist2_final >= (Threshold_dist * speed_distance_sens_scaling))
130 {
131 countSendStop_sens2=0;
132 countSendStart_sens2++;
133 i2c_command.command = I2C_START_MOTOR;
134 i2c_command.commandValue = 0;
135 i2c_command.sendingSensor = I2C_distance_sens2_dev_handle;
136 xQueueSend(I2C_commandQueue,&i2c_command,-1);
137 }
138 // Log the filtered value
139 //ESP_LOGI("I2C", "Range2: [ %.2lf ]", readValSensor_dist2_final);
140 break;
141
142 case I2C_adxl345_sens_mux: // if
143 double x, y, z;
144 static double roll_init_offset = 0, pitch_init_offset = 0, count = 0;
145 bool init_offset = false;
146 I2C_read_adxl345_data(current_sensor->device_handle, &x, &y, &z);
147
148 // Calculate roll and pitch
149 double roll = atan2(y, sqrt(x * x + z * z)) * 180.0 / M_PI;
150 double pitch = atan2(-x, sqrt(y * y + z * z)) * 180.0 / M_PI;
151 static double roll_prev=0, pitch_prev=0, roll_final = 0, pitch_final = 0;
152 if(init_offset == false)
153 {
154 count++;
155 if(count<6)
156 {
157 init_offset = true;
158 roll_init_offset = roll;
159 pitch_init_offset = pitch;
160 }
161 }
162 roll -=roll_init_offset;
163 pitch -=pitch_init_offset;
164
165 /* Apply low pass filter */
166 roll_final = ALPHA_ADXL * (double)roll + (1 - ALPHA_ADXL) * (double)roll_prev; // Update the previous value for the next iteration
167 roll_prev = roll;
168
169 pitch_final = ALPHA_ADXL * (double)pitch + (1 - ALPHA_ADXL) * (double)pitch_prev; // Update the previous value for the next iteration
170 pitch_prev = pitch;
171
172 //ESP_LOGI("I2C", "Orientation roll=%lf pitch=%lf ", roll_final, pitch_final);
173
174 sendCommandApp(ADXL_ROLL, (double*)&roll_final, DOUBLE);
175 sendCommandApp(ADXL_PITCH, (double*)&pitch_final, DOUBLE);
176
177 break;
178
179 case I2C_pixy2_camera_mux:
180#if (PIXY_DETECTION == ON)
181 Vector goodVecLeft = {0};
182 Vector goodVecRight = {0};
183 pixy2Commands commandAutonomous;
184 bool left = false;
185 bool right = false;
186 getPixy2Lines(I2C_pixy2_dev_handle, LINE_VECTOR , true, &lines); // Get all features (vectors, intersections, barcodes) with wait
187 left = getBestVectorLeft(&lines, &goodVecLeft);
188 right = getBestVectorRight(&lines, &goodVecRight);
189 computeSpeedAndSteer(goodVecLeft, goodVecRight, left, right, &commandAutonomous);
190if(AutonomousMode == true){
191 if (xQueueSend(autonomousModeControlPixyQueue, &commandAutonomous.computedSpeedSetpoint, portMAX_DELAY) != pdPASS) {
192 ESP_LOGE("Error uart", "Unable to send to queue");
193 }
194 if (xQueueSend(autonomousModeControlPixyQueue, &commandAutonomous.computedSteerSetpoint, portMAX_DELAY) != pdPASS) {
195 ESP_LOGE("Error uart", "Unable to send to queue");
196 }
197}
198
199#endif
200 break;
201
202 case I2C_temp_sens_mux:
203 I2C_read_temperature(&temp);
204 //ESP_LOGI("I2C", "Temp: [ %lf ]", temp);
205 sendCommandApp(TEMPERATURE, (double*)&temp, DOUBLE);
206 break;
207
208 default:
209 break;
210 }
211 current_sensor->state = SENSOR_READY;
212 break;
213
214 case SENSOR_READY:
215 if (--current_sensor->tokens_remaining == 0) {
216 current_sensor->tokens_remaining = current_sensor->tokens;
217 current_sensor->state = SENSOR_IDLE;
218 current_sensor_index = (current_sensor_index + 1) % sensor_count;
219 current_sensor = &sensors[current_sensor_index];
220 } else {
221 current_sensor->state = SENSOR_IDLE;
222 }
223 break;
224 }
225 vTaskDelay(pdMS_TO_TICKS(1));
226 }

230bool I2C_addAndInitialiseSensors()
231{
232 config_rst_pin_i2c_mux(); // config rst pin for the mux
233 rst_pin_i2c_mux_off(); // rst pin on
234 vTaskDelay(pdMS_TO_TICKS(2000));
235 rst_pin_i2c_mux_on(); // rst pin on
236
237 //Add sensors
238 I2C_add_device(I2C_mux_addr);// add the multiplexer to the I2C bus
239 I2C_add_device(I2C_temp_sens_addr);
240 I2C_add_device(I2C_distance_sens_addr);
241 I2C_add_device(I2C_adxl345_sens_addr);
242 I2C_add_device(I2C_pixy2_camera_addr);
243
244 //Check Pixy2 camera version
245 requestPixy2Version(I2C_pixy2_dev_handle);
246
247 //Init sensors
248 I2C_adxl345_init(I2C_adxl345_sens_dev_handle);
249 I2C_adxl345_setRate(I2C_adxl345_sens_dev_handle, ADXL1345_READ_RATE);
250 I2C_temperatureSensor_trigger_measurement();
251 bool dist_sens_init = VL53L0X_Init(I2C_distance_sens2_dev_handle);
252 while(dist_sens_init == false) {vTaskDelay(pdMS_TO_TICKS(50));}
253 dist_sens_init = false;
254 dist_sens_init = VL53L0X_Init(I2C_distance_sens1_dev_handle);
255 while(dist_sens_init == false) {vTaskDelay(pdMS_TO_TICKS(50));}
256
257 return true;
258}
