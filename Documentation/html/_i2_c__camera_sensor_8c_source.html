100void getPixy2Lines(I2C_dev_handles pixy2_handle, uint8_t features, bool wait, Pixy2Lines *lines) {
101 uint8_t requestPacket[6] = {0xae, 0xc1, 48, 2, features, wait};
102
103 // Select the Pixy2 camera channel on the multiplexer
104 I2C_select_multiplexer_channel(I2C_pixy2_camera_mux);
105
106 // Transmit the request packet to Pixy2
107 I2C_transmit(pixy2_handle, requestPacket, sizeof(requestPacket));
108 //ESP_LOGI(TAG, "Sent get main features request to Pixy2");
109
110 // Prepare to receive the response
111 uint8_t responsePacket[100]; // Adjust size based on expected response length
112
113 // Receive the response packet from Pixy2
114 I2C_receive(pixy2_handle, responsePacket, sizeof(responsePacket));
115
116 // Initialize the number of vectors to 0
117 lines->numVectors = 0;
118
119 // Validate sync and packet type
120 if (responsePacket[0] == 0xaf && responsePacket[1] == 0xc1 && responsePacket[2] == 49) {
121 uint8_t payloadLength = responsePacket[3]; // payload length is 1 byte
122
123 // Calculate the total length to read
124 int totalLength = 4 + payloadLength; // Includes the headers before payload
125
126 int index = 6; // Start after the checksum (bytes 4-5)
127
128 while (index < totalLength) {
129 uint8_t featureType = responsePacket[index];
130 uint8_t featureLength = responsePacket[index + 1];
131 uint8_t* featureData = &responsePacket[index + 2];
132
133 // Loop through each vector in this feature
134 if (featureType == LINE_VECTOR) {
135 int numberOfVectors = featureLength / 6; // Each vector is 6 bytes
136 for (int i = 0; i < numberOfVectors; i++) {
137 int offset = i * 6;
138 uint8_t x0 = featureData[offset];
139 uint8_t y0 = featureData[offset + 1];
140 uint8_t x1 = featureData[offset + 2];
141 uint8_t y1 = featureData[offset + 3];
142 uint8_t vectorIndex = featureData[offset + 4];
143
144 if (lines->numVectors < MAX_VECTORS) {
145 lines->vectors[lines->numVectors].x0 = x0;
146 lines->vectors[lines->numVectors].y0 = HEIGHT - y0; // The vectors have the coordinate starting in the top-left corner.
147 lines->vectors[lines->numVectors].x1 = x1;
148 lines->vectors[lines->numVectors].y1 = HEIGHT - y1;
149 lines->vectors[lines->numVectors].index = vectorIndex;
150 lines->numVectors++;
151 }
152 }
153 }
154 index += 2 + featureLength; // Move to the next feature
155 }
156 } else {
157 //ESP_LOGI(TAG, "Invalid response from Pixy2");
158 }
159}

161bool getBestVectorLeft(Pixy2Lines *lines, Vector *vec) {
162 bool found = false;
163 double maxLength = 0;
164 double minSlopeThreshold = 0.35; // Lowered threshold to include nearly horizontal lines
165
166 for (int i = 0; i < lines->numVectors; i++) {
167 Vector v = lines->vectors[i];
168 double midpoint = (v.x0 + v.x1) / 2.0;
169 if (midpoint <= 39) {
170 double dx = (double)(v.x1 - v.x0);
171 double dy = (double)(v.y1 - v.y0);
172 double slope = dx == 0 ? INFINITY : dy / dx; // Handle vertical lines safely
173 double length = sqrt(dx * dx + dy * dy);
174 double absSlope = fabs(slope);
175
176 // Consider vectors with very low slope if they are long enough
177 if ((absSlope >= minSlopeThreshold || length > 50) && length > maxLength) { // Arbitrary length threshold
178 maxLength = length;
179 *vec = v;
180 found = true;
181 }
182 }
183 }
184 return found;

215void computeSpeedAndSteer(Vector vecLeft, Vector vecRight, bool existsGoodVecLeft, bool existsGoodVecRight, pixy2Commands *commands) {
216 static uint8_t lastSteer = 50; // Neutral position
217 static uint8_t lastSpeed = 0;
218 float frameWidth = 78.0; // Frame width assumption
219 float frameHeight = 52.0;
220 float frame_center = 39.0; // Central position of the frame to respect
221 // Determine number of good vectors detected
222 int numVectors = (existsGoodVecLeft ? 1 : 0) + (existsGoodVecRight ? 1 : 0);
223
224 switch (numVectors) {
225 case 0:
226 // No vectors detected
227 commands->computedSteerSetpoint = lastSteer;
228 commands->computedSpeedSetpoint = (uint8_t)(0.7 * 30); // Reduce speed significantly
229 break;
230
231 case 1:
232 {
233 Vector single_vec;
234 if (existsGoodVecRight) {
235 single_vec = vecRight;
236 } else {
237 single_vec = vecLeft;
238 }
239
240 // Calculate the slope of the detected vector
241 float single_vec_slope = (float)(single_vec.y1 - single_vec.y0) / (float)(single_vec.x1 - single_vec.x0);
242 float single_vec_midpoint_x = (float)(single_vec.x1 + single_vec.x0) / 2;
243
244 float estimated_lane_width = 78.0f;
245 float estimated_other_boundary_midpoint_x;
246
247 if (single_vec_slope > 0) {
248 estimated_other_boundary_midpoint_x = single_vec_midpoint_x + estimated_lane_width;
249 } else {
250 estimated_other_boundary_midpoint_x = single_vec_midpoint_x - estimated_lane_width;
251 }
252
253 float lane_center_x = (single_vec_midpoint_x + estimated_other_boundary_midpoint_x) / 2;
254
255 float error = (lane_center_x - frame_center) / frame_center;
256
257 // Implement proportional-derivative control for steering
258 float Kp = 0.8f; // Proportional gain (adjust as needed)
259 float Kd = 0.2f; // Derivative gain (adjust as needed)
260
261 static float last_error = 0;
262 float derivative = error - last_error;
263 float control_steer = -(Kp * error + Kd * derivative);
264 //float control_steer = - Kp * error;
265 // Update last_error
266 last_error = error;
267
268 // Make sure the steering angle is within the range [0, 100]
269 float steer = 50 + control_steer * 50; // Map control_steer from [-1, 1] to [0, 100]
270 commands->computedSteerSetpoint = (uint8_t)fmax(0, fmin(100, steer));
271
272 break;
273 }
274
275 case 2: // Both vectors detected, calculate direct center
276 {
277 float MidpointX0 = (vecRight.x0 + vecLeft.x0) / 2.0;
278 float MidpointX1 = (vecRight.x1 + vecLeft.x1) / 2.0;
279 float MidpointY0 = (vecRight.y0 + vecLeft.y0) / 2.0;
280 float MidpointY1 = (vecRight.y1 + vecLeft.y1) / 2.0;
281
282 float new_x = (MidpointX1 - MidpointX0) / frameWidth;
283 float new_y = (MidpointY1 - MidpointY0) / frameHeight;
284
285 if (fabs(new_y) > 0.01f) { // Avoid division by zero
286 float gradient = new_x / new_y; // Calculate gradient
287 // Normalize the gradient to fit within the range of [-1, 1]
288 gradient = fmax(fmin(gradient, 1), -1); // Clamping the value to [-1, 1] if out of bounds
289 // Map gradient to steering setpoint range [0, 100]
290 commands->computedSteerSetpoint = (uint8_t)(50 - gradient * 50);
291 } else {
292 // Go straight
293 commands->computedSteerSetpoint = 50;
294 }
295
296 break;
297 }
298 }
299
300 // Adjust speed based on steer: more deviation from 50, lower the speed
301 float steer_deviation = fabs(50.0f - (float)commands->computedSteerSetpoint);
302 commands->computedSpeedSetpoint = (uint8_t)(60 - (steer_deviation / 50.0f) * (60 - 25));
303
304 // Ensure the speed is within the allowable range
305 if (commands->computedSpeedSetpoint > 65) commands->computedSpeedSetpoint = 65;
306 //if (commands->computedSpeedSetpoint < 0) commands->computedSpeedSetpoint = 0;
307
308 lastSteer = commands->computedSteerSetpoint; // Update last known steer
309 lastSpeed = commands->computedSpeedSetpoint;
310}
