12bool VL53L0X_getSpadInfo(I2C_dev_handles device_handle ,uint8_t * count, bool * type_is_aperture)
13{
14 uint8_t tmp;
15
16 I2C_writeRegister8bit(device_handle, 0x80, 0x01);
17 I2C_writeRegister8bit(device_handle, 0xFF, 0x01);
18 I2C_writeRegister8bit(device_handle, 0x00, 0x00);
19 I2C_writeRegister8bit(device_handle, 0xFF, 0x06);
20 I2C_writeRegister8bit(device_handle, 0x83, (uint8_t)(I2C_readRegister16bit(device_handle, 0x83)>>8) | 0x04);
21 I2C_writeRegister8bit(device_handle, 0xFF, 0x07);
22 I2C_writeRegister8bit(device_handle, 0x81, 0x01);
23 I2C_writeRegister8bit(device_handle, 0x80, 0x01);
24 I2C_writeRegister8bit(device_handle, 0x94, 0x6b);
25 I2C_writeRegister8bit(device_handle, 0x83, 0x00);
26 uint8_t reg;
27 startTimeout();
28 do
29 {
30 reg = (uint8_t)(I2C_readRegister16bit(device_handle, 0x83)>>8);
31 if (checkTimeoutExpired()) { return false; }
32 vTaskDelay(pdMS_TO_TICKS(10));
33 } while(reg == 0x00);
34 I2C_writeRegister8bit(device_handle, 0x83, 0x01);
35 tmp = (uint8_t)(I2C_readRegister16bit(device_handle, 0x92)>>8);
36
37 *count = tmp & 0x7f;
38 *type_is_aperture = (tmp >> 7) & 0x01;
39
40 I2C_writeRegister8bit(device_handle, 0x81, 0x00);
41 I2C_writeRegister8bit(device_handle, 0xFF, 0x06);
42// uint8_t data;
43 I2C_writeRegister8bit(device_handle, 0x83, (uint8_t)(I2C_readRegister16bit(device_handle, 0x83)>>8) & ~0x04);
44 I2C_writeRegister8bit(device_handle, 0xFF, 0x01);
45 I2C_writeRegister8bit(device_handle, 0x00, 0x01);
46
47 I2C_writeRegister8bit(device_handle, 0xFF, 0x00);
48 I2C_writeRegister8bit(device_handle, 0x80, 0x00);
49
50 return true;
51}

58static type_error VL53L0X_setSignalRateLimit (I2C_dev_handles device_handle, float limit_Mcps)
59{
60 if (limit_Mcps < 0 || limit_Mcps > 511.99)
61 return err;
62 // Q9.7 fixed point format (9 integer bits, 7 fractional bits)
63 I2C_writeRegister16bit(device_handle, FINAL_RANGE_CONFIG_MIN_COUNT_RATE_RTN_LIMIT, limit_Mcps * (1 << 7));
64 return not_err;
65}

67bool VL53L0X_Init(I2C_dev_handles device_handle)
68{
69 if((uint8_t)(I2C_readRegister16bit(device_handle, IDENTIFICATION_MODEL_ID)>>8)!=0xEE)
70 return false;
71
72 uint16_t data;
73
74 // "Set I2C standard mode"
75 I2C_writeRegister8bit(device_handle,0x88, 0x00);
76 I2C_writeRegister8bit(device_handle,0x80, 0x01);
77 I2C_writeRegister8bit(device_handle,0xFF, 0x01);
78 I2C_writeRegister8bit(device_handle,0x00, 0x00);
79
80 stop_variable = (uint8_t)(I2C_readRegister16bit(device_handle, 0x91)>>8);
81 ESP_LOGI(" ", "stop_variable: %u", stop_variable);
82 I2C_writeRegister8bit(device_handle,0x00, 0x01);
83 I2C_writeRegister8bit(device_handle,0xFF, 0x00);
84 I2C_writeRegister8bit(device_handle,0x80, 0x00);
85
86 // disable SIGNAL_RATE_MSRC (bit 1) and SIGNAL_RATE_PRE_RANGE (bit 4) limit checks
87 uint8_t MSRC_CONFIG = (uint8_t)(I2C_readRegister16bit(device_handle, MSRC_CONFIG_CONTROL)>>8);
88 I2C_writeRegister8bit(device_handle, MSRC_CONFIG_CONTROL, MSRC_CONFIG | 0x12) ;
89
90
91
92 data = VL53L0X_setSignalRateLimit (device_handle, 0.25);
93
94
95 I2C_writeRegister8bit(device_handle, SYSTEM_SEQUENCE_CONFIG, 0xFF);
96
97 uint8_t spad_count;
98 bool spad_type_is_aperture;
99 if (!VL53L0X_getSpadInfo(device_handle ,&spad_count, &spad_type_is_aperture)) {
100 //ESP_LOGI(" ","VL53L0X_getSpadInfo false");
101 return false;
102 }
103 uint8_t ref_spad_map[6];
104 uint8_t spad_enables_ref_0 = GLOBAL_CONFIG_SPAD_ENABLES_REF_0;
105
106 I2C_transmit(device_handle, &spad_enables_ref_0, 1);
107 vTaskDelay(pdMS_TO_TICKS(10));
108 I2C_receive(device_handle, ref_spad_map, 6);
109 vTaskDelay(pdMS_TO_TICKS(10));
110
111 I2C_writeRegister8bit(device_handle, 0xFF, 0x01);
112 I2C_writeRegister8bit(device_handle, DYNAMIC_SPAD_REF_EN_START_OFFSET, 0x00);
113 I2C_writeRegister8bit(device_handle, DYNAMIC_SPAD_NUM_REQUESTED_REF_SPAD, 0x2C);
114 I2C_writeRegister8bit(device_handle, 0xFF, 0x00);
115 I2C_writeRegister8bit(device_handle, GLOBAL_CONFIG_REF_EN_START_SELECT, 0xB4);
116
117 uint8_t first_spad_to_enable = spad_type_is_aperture ? 12 : 0; // 12 is the first aperture spad
118 uint8_t spads_enabled = 0;
119
120 for (uint8_t i = 0; i < 48; i++)
121 {
122 if (i < first_spad_to_enable || spads_enabled == spad_count)
123 {
124 // This bit is lower than the first one that should be enabled, or
125 // (reference_spad_count) bits have already been enabled, so zero this bit
126 ref_spad_map[i / 8] &= ~(1 << (i % 8));
127 }
128 else if ((ref_spad_map[i / 8] >> (i % 8)) & 0x1)
129 {
130 spads_enabled++;
131 }
132 }
133 I2C_transmit(device_handle, &spad_enables_ref_0, 1);
134 vTaskDelay(pdMS_TO_TICKS(10));
135 I2C_transmit(device_handle, ref_spad_map, sizeof(ref_spad_map)); // 6
136
137 VL53L0X_setupI2CRegisters(device_handle);
138
139 I2C_writeRegister8bit(device_handle, SYSTEM_INTERRUPT_CONFIG_GPIO, 0x04);
140 I2C_writeRegister8bit(device_handle, GPIO_HV_MUX_ACTIVE_HIGH, (uint8_t)(I2C_readRegister16bit(device_handle, GPIO_HV_MUX_ACTIVE_HIGH)>>8) & ~0x10); // active low
141 I2C_writeRegister8bit(device_handle, SYSTEM_INTERRUPT_CLEAR, 0x01);
142
143 measurement_timing_budget_us = VL53L0X_getMeasurementTimingBudget(device_handle);
144 ESP_LOGI(" ","VL53L0X_setMeasurementTimingBudget %"PRIu32" \n", measurement_timing_budget_us);
145 I2C_writeRegister8bit(device_handle, SYSTEM_SEQUENCE_CONFIG, 0xE8);
146 if (!VL53L0X_setMeasurementTimingBudget(device_handle,measurement_timing_budget_us-10)) {
147 //ESP_LOGI(" ","VL53L0X_setMeasurementTimingBudget false \n");
148 return false;
149 };
150 I2C_writeRegister8bit(device_handle, SYSTEM_SEQUENCE_CONFIG, 0x01);
151 if (!VL53L0X_performSingleRefCalibration(device_handle, 0x40)) {
152 //ESP_LOGI(" ","VL53L0X_performSingleRefCalibration1 false \n");
153 return false;
154 }
155 I2C_writeRegister8bit(device_handle, SYSTEM_SEQUENCE_CONFIG, 0x02);
156 if (!VL53L0X_performSingleRefCalibration(device_handle, 0x00)) {
157 // ESP_LOGI(" ","VL53L0X_performSingleRefCalibration2 false \n");
158 return false;
159 }
160
161 I2C_writeRegister8bit(device_handle, SYSTEM_SEQUENCE_CONFIG, 0xE8);
162
163 //ESP_LOGI(" ","VLX sensor init done! \n");
164 return true;
165}

167uint32_t VL53L0X_getMeasurementTimingBudget(I2C_dev_handles device_handle)
168{
169 SequenceStepEnables enables;
170 SequenceStepTimeouts timeouts;
171
172 uint16_t const StartOverhead = 1910;
173 uint16_t const EndOverhead = 960;
174 uint16_t const MsrcOverhead = 660;
175 uint16_t const TccOverhead = 590;
176 uint16_t const DssOverhead = 690;
177 uint16_t const PreRangeOverhead = 660;
178 uint16_t const FinalRangeOverhead = 550;
179
180 // "Start and end overhead times always present"
181 uint32_t budget_us = StartOverhead + EndOverhead;
182
183 VL53L0X_getSequenceStepEnables(device_handle , &enables);
184 VL53L0X_getSequenceStepTimeouts(device_handle, &enables, &timeouts);
185
186 if (enables.tcc)
187 {
188 budget_us += (timeouts.msrc_dss_tcc_us + TccOverhead);
189 }
190
191 if (enables.dss)
192 {
193 budget_us += 2 * (timeouts.msrc_dss_tcc_us + DssOverhead);
194 }
195 else if (enables.msrc)
196 {
197 budget_us += (timeouts.msrc_dss_tcc_us + MsrcOverhead);
198 }
199
200 if (enables.pre_range)
201 {
202 budget_us += (timeouts.pre_range_us + PreRangeOverhead);
203 }
204
205 if (enables.final_range)
206 {
207 budget_us += (timeouts.final_range_us + FinalRangeOverhead);
208 }
209
210 measurement_timing_budget_us = budget_us; // store for internal reuse
211 return budget_us;
212}

214bool VL53L0X_setMeasurementTimingBudget(I2C_dev_handles device_handle, uint32_t budget_us)
215{
216 SequenceStepEnables enables;
217 SequenceStepTimeouts timeouts;
218
219 uint16_t const StartOverhead = 1910;
220 uint16_t const EndOverhead = 960;
221 uint16_t const MsrcOverhead = 660;
222 uint16_t const TccOverhead = 590;
223 uint16_t const DssOverhead = 690;
224 uint16_t const PreRangeOverhead = 660;
225 uint16_t const FinalRangeOverhead = 550;
226
227 uint32_t used_budget_us = StartOverhead + EndOverhead;
228
229 VL53L0X_getSequenceStepEnables(device_handle,&enables);
230 VL53L0X_getSequenceStepTimeouts(device_handle,&enables, &timeouts);
231
232 if (enables.tcc)
233 {
234 used_budget_us += (timeouts.msrc_dss_tcc_us + TccOverhead);
235 }
236
237 if (enables.dss)
238 {
239 used_budget_us += 2 * (timeouts.msrc_dss_tcc_us + DssOverhead);
240 }
241 else if (enables.msrc)
242 {
243 used_budget_us += (timeouts.msrc_dss_tcc_us + MsrcOverhead);
244 }
245
246 if (enables.pre_range)
247 {
248 used_budget_us += (timeouts.pre_range_us + PreRangeOverhead);
249 }
250
251 if (enables.final_range)
252 {
253 used_budget_us += FinalRangeOverhead;
254
255 // "Note that the final range timeout is determined by the timing
256 // budget and the sum of all other timeouts within the sequence.
257 // If there is no room for the final range timeout, then an error
258 // will be set. Otherwise the remaining time will be applied to
259 // the final range."
260
261 if (used_budget_us > budget_us)
262 {
263 // "Requested timeout too big."
264 return false;
265 }
266
267 uint32_t final_range_timeout_us = budget_us - used_budget_us;
268 // (SequenceStepId == VL53L0X_SEQUENCESTEP_FINAL_RANGE)
269
270 // "For the final range timeout, the pre-range timeout
271 // must be added. To do this both final and pre-range
272 // timeouts must be expressed in macro periods MClks
273 // because they have different vcsel periods."
274 uint32_t final_range_timeout_mclks =
275 VL53L0X_timeoutMicrosecondsToMclks(final_range_timeout_us,
276 timeouts.final_range_vcsel_period_pclks);
277
278 if (enables.pre_range)
279 {
280 final_range_timeout_mclks += timeouts.pre_range_mclks;
281 }
282
283 I2C_writeRegister8bit(device_handle, FINAL_RANGE_CONFIG_TIMEOUT_MACROP_HI,
284 VL53L0X_encodeTimeout(final_range_timeout_mclks));
285
286 // set_sequence_step_timeout() end
287
288 measurement_timing_budget_us = budget_us; // store for internal reuse
289 }
290 return true;
291}

294void VL53L0X_getSequenceStepEnables(I2C_dev_handles device_handle, SequenceStepEnables * enables)
295{
296 uint8_t sequence_config = (uint8_t)(I2C_readRegister16bit(device_handle, SYSTEM_SEQUENCE_CONFIG)>>8);
297
298 enables->tcc = (sequence_config >> 4) & 0x1;
299 enables->dss = (sequence_config >> 3) & 0x1;
300 enables->msrc = (sequence_config >> 2) & 0x1;
301 enables->pre_range = (sequence_config >> 6) & 0x1;
302 enables->final_range = (sequence_config >> 7) & 0x1;
303}

305void VL53L0X_getSequenceStepTimeouts(I2C_dev_handles device_handle, SequenceStepEnables const * enables, SequenceStepTimeouts * timeouts)
306{
307 timeouts->pre_range_vcsel_period_pclks = VL53L0X_getVcselPulsePeriod(device_handle, VcselPeriodPreRange);
308 timeouts->msrc_dss_tcc_mclks = (uint8_t)(I2C_readRegister16bit(device_handle, MSRC_CONFIG_TIMEOUT_MACROP)>>8) + 1;
309 timeouts->msrc_dss_tcc_us = VL53L0X_timeoutMclksToMicroseconds(timeouts->msrc_dss_tcc_mclks,
310 timeouts->pre_range_vcsel_period_pclks);
311 timeouts->pre_range_mclks = VL53L0X_decodeTimeout(I2C_readRegister16bit(device_handle, PRE_RANGE_CONFIG_TIMEOUT_MACROP_HI));
312 timeouts->pre_range_us =VL53L0X_timeoutMclksToMicroseconds(timeouts->pre_range_mclks,
313 timeouts->pre_range_vcsel_period_pclks);
314 timeouts->final_range_vcsel_period_pclks = VL53L0X_getVcselPulsePeriod(device_handle, VcselPeriodFinalRange);
315 timeouts->final_range_mclks =
316 VL53L0X_decodeTimeout(I2C_readRegister16bit(device_handle, FINAL_RANGE_CONFIG_TIMEOUT_MACROP_HI));
317
318 if (enables->pre_range)
319 {
320 timeouts->final_range_mclks -= timeouts->pre_range_mclks;
321 }
322 timeouts->final_range_us =
323 VL53L0X_timeoutMclksToMicroseconds(timeouts->final_range_mclks,
324 timeouts->final_range_vcsel_period_pclks);
325}

327// Get the VCSEL pulse period in PCLKs for the given period type.
328// based on VL53L0X_get_vcsel_pulse_period()
329uint8_t VL53L0X_getVcselPulsePeriod(I2C_dev_handles device_handle, vcselPeriodType type)
330{
331 if (type == VcselPeriodPreRange)
332 {
333 return decodeVcselPeriod((uint8_t)(I2C_readRegister16bit(device_handle, PRE_RANGE_CONFIG_VCSEL_PERIOD)>>8));
334 }
335 else if (type == VcselPeriodFinalRange)
336 {
337 return decodeVcselPeriod((uint8_t)(I2C_readRegister16bit(device_handle, FINAL_RANGE_CONFIG_VCSEL_PERIOD)>>8));
338 }
339 else { return 255; }
340}

349uint16_t VL53L0X_decodeTimeout(uint16_t reg_val)
350{
351 // format: "(LSByte * 2^MSByte) + 1"
352 return (uint16_t)((reg_val & 0x00FF) <<
353 (uint16_t)((reg_val & 0xFF00) >> 8)) + 1;
354}

356uint32_t VL53L0X_timeoutMicrosecondsToMclks(uint32_t timeout_period_us, uint8_t vcsel_period_pclks)
357{
358 uint32_t macro_period_ns = calcMacroPeriod(vcsel_period_pclks);
359 return (((timeout_period_us * 1000) + (macro_period_ns / 2)) / macro_period_ns);
360}

362uint16_t VL53L0X_encodeTimeout(uint32_t timeout_mclks)
363{
364 // format: "(LSByte * 2^MSByte) + 1"
365
366 uint32_t ls_byte = 0;
367 uint16_t ms_byte = 0;
368
369 if (timeout_mclks > 0)
370 {
371 ls_byte = timeout_mclks - 1;
372
373 while ((ls_byte & 0xFFFFFF00) > 0)
374 {
375 ls_byte >>= 1;
376 ms_byte++;
377 }
378
379 return (ms_byte << 8) | (ls_byte & 0xFF);
380 }
381 else { return 0; }
382}

430uint16_t VL53L0X_readRangeContinuousMillimeters(I2C_dev_handles device_handle)
431{
432 startTimeout();
433 uint16_t range = 0;
434 static bool start_continuous_back2back_sens1 = false;
435 static bool start_continuous_back2back_sens2 = false;
436 if((start_continuous_back2back_sens1 == false && device_handle == I2C_distance_sens1_dev_handle) ||
437 ((start_continuous_back2back_sens2 == false && device_handle == I2C_distance_sens2_dev_handle)))
438 {
439 I2C_writeRegister8bit(device_handle, 0x80, 0x01);
440 I2C_writeRegister8bit(device_handle, 0xFF, 0x01);

447 if(device_handle == I2C_distance_sens1_dev_handle)
448 start_continuous_back2back_sens1 = true;
449 else start_continuous_back2back_sens2 = true;
450 vTaskDelay(pdMS_TO_TICKS(20));

452
453 while (((uint8_t)(I2C_readRegister16bit(device_handle, RESULT_INTERRUPT_STATUS)>>8) & 0x07) == 0)

455 if (checkTimeoutExpired())
456 {
457 //did_timeout = true;

459 }

464 I2C_writeRegister8bit(device_handle, SYSTEM_INTERRUPT_CLEAR, 0x01);
465
466
