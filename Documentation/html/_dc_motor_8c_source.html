15// Initialize PWM for Motor
16void init_motor_pwm() {
17#if (MOTOR_CONTROL == LEDC)
18 // Timer configuration
19 ledc_timer_config_t ledc_timer =
20 {
21 .speed_mode = LEDC_Motor_And_Servo_MODE,
22 .duty_resolution = LEDC_MOTOR_DUTY_RESOLUTION,
23 .timer_num = LEDC_MOTOR_TIMER, // Different timer for motor
24 .freq_hz = LEDC_Motor_And_Servo_FREQ,
25 };
26 ledc_timer_config(&ledc_timer);
27
28 // Channel configuration, similar to steering but different channel and GPIO
29 ledc_channel_config_t ledc_channel =
30 {
31 .channel = LEDC_MOTOR_CH1_CHANNEL, // Different channel for motor
32 .duty = 0,
33 .gpio_num = LEDC_MOTOR_GPIO, // Different GPIO for motor
34 .speed_mode = LEDC_Motor_And_Servo_MODE,
35 .timer_sel = LEDC_MOTOR_TIMER, // Different timer for motor
36 };
37 ledc_channel_config(&ledc_channel);
38#elif (MOTOR_CONTROL == MCPWM)
39
40 // Configure MCPWM timer
41 mcpwm_timer_config_t config =
42 {
43 .group_id = 1,
44 //.intr_priority = 0,
45 .clk_src = MCPWM_TIMER_CLK_SRC_DEFAULT,
46 // period_ticks=Desired Period * Resolution
47 // period_ticks= 20000 microseconds * 1 microsecond
48 .resolution_hz = 1000000, // 1MHZ => 1 tick = 1/1000000 = 1 microsecond each tick
49 .period_ticks = 20000, // 20000 * 1 microsecond = 20000 microseconds => 0.02 seconds => 50 hz
50 .count_mode = MCPWM_TIMER_COUNT_MODE_UP,
51 };
52
53 ESP_ERROR_CHECK(mcpwm_new_timer(&config, &timer_dc_motor));
54
55 // Configure MCPWM operator
56 mcpwm_operator_config_t operator_config = {
57 .group_id = 1,
58 .flags.update_gen_action_on_tez = 1, // Update generator action when timer counts to zero
59 };
60
61 ESP_ERROR_CHECK(mcpwm_new_operator(&operator_config, &operator_dc_motor));
62 ESP_ERROR_CHECK(mcpwm_operator_connect_timer(operator_dc_motor, timer_dc_motor));
63
64 // Configure MCPWM comparator
65 mcpwm_comparator_config_t comparator_config =
66 {
67 .flags.update_cmp_on_tez = 1,
68 };
69
70 ESP_ERROR_CHECK(mcpwm_new_comparator(operator_dc_motor, &comparator_config, &comparator_dc_motor));
71
72 // Configure MCPWM generator
73 mcpwm_generator_config_t generator_config = {
74 .gen_gpio_num = LEDC_MOTOR_GPIO, // Set the GPIO number used by the generator
75 };
76
77 ESP_ERROR_CHECK(mcpwm_new_generator(operator_dc_motor, &generator_config, &generator_dc_motor));
78
79 // Set the generator actions
80 // go high on counter empty
81 ESP_ERROR_CHECK(mcpwm_generator_set_action_on_timer_event(generator_dc_motor,
82 MCPWM_GEN_TIMER_EVENT_ACTION(MCPWM_TIMER_DIRECTION_UP, MCPWM_TIMER_EVENT_EMPTY, MCPWM_GEN_ACTION_HIGH)));
83 // go low on compare threshold
84 ESP_ERROR_CHECK(mcpwm_generator_set_action_on_compare_event(generator_dc_motor,
85 MCPWM_GEN_COMPARE_EVENT_ACTION(MCPWM_TIMER_DIRECTION_UP, comparator_dc_motor, MCPWM_GEN_ACTION_LOW)));
86
87 // Set the duty cycle for the servo
88 ESP_ERROR_CHECK(mcpwm_comparator_set_compare_value(comparator_dc_motor, 1500)); // 1.5ms pulse width for neutral position
89
90 // Start the MCPWM timer
91 ESP_ERROR_CHECK(mcpwm_timer_enable(timer_dc_motor));
92 ESP_ERROR_CHECK(mcpwm_timer_start_stop(timer_dc_motor, MCPWM_TIMER_START_NO_STOP));
93
94
95#endif
96}

98// Update the PWM for Motor, similar to update_servo_pwm function
99void update_motor_pwm(unsigned int pulse_width_us) {
100#if (MOTOR_CONTROL == LEDC)
101 uint32_t duty = (pulse_width_us * ((1 << LEDC_MOTOR_DUTY_RESOLUTION) - 1))
102 / 20000;
103 ledc_set_duty(LEDC_Motor_And_Servo_MODE, LEDC_MOTOR_CH1_CHANNEL, duty);
104 ledc_update_duty(LEDC_Motor_And_Servo_MODE, LEDC_MOTOR_CH1_CHANNEL);
105#elif (MOTOR_CONTROL == MCPWM)
106 ESP_ERROR_CHECK(mcpwm_comparator_set_compare_value(comparator_dc_motor, pulse_width_us));
107 //ESP_LOGI("pulse_width_us", "us = %d", pulse_width_us);
108#endif
109}

111/* Change Motor Speed, similar to changeSTEER function */
112void changeMotorSpeed(int value) {
113 int pulse_width_us;
114 //ESP_LOGI("", "value %d", value);
115 if(value > 100)
116 value = 100;
117 else if(value < -100)
118 value = -100;
119
120 if (value == 0)
121 pulse_width_us = 1500;
122 else if (value >= 1) { // FORWARD
123 // Forward mode: Scale between 1545 and 1700
124 pulse_width_us = MIN_MOTOR_FW_DUTY_US + (value * (MAX_MOTOR_FW_DUTY_US - MIN_MOTOR_FW_DUTY_US) / 100);
125 //ESP_LOGI("", "pwm_us %d", pulse_width_us);
126 } else { // BACKWARD
127 pulse_width_us = MAX_MOTOR_BW_DUTY_US - ((-value) * (MAX_MOTOR_BW_DUTY_US - MIN_MOTOR_BW_DUTY_US) / 100);
128 //ESP_LOGI("", "pwm_us %d", pulse_width_us);
129 }
130 update_motor_pwm(pulse_width_us);
131}
