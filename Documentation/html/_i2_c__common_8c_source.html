236uint32_t I2C_readRegister32bit(I2C_dev_handles device_handle, uint8_t reg)
237{
238 uint8_t data[4]; // Array to hold the four bytes read from the device
239 uint32_t data_f = 0; // Initialize the final 32-bit data value
240
241 // Send the register address from which to start the read
242 I2C_transmit(device_handle, &reg, 1);
243 vTaskDelay(pdMS_TO_TICKS(50)); // Delay to allow for address setup
244
245 // Receive 4 bytes of data
246 I2C_receive(device_handle, data, 4);
247 vTaskDelay(pdMS_TO_TICKS(50)); // Delay to allow the read transaction to complete
248
249 // Convert the 4 bytes into a single 32-bit integer (assume big-endian order)
250 data_f = (uint32_t)data[0] << 24; // Highest byte
251 data_f |= (uint32_t)data[1] << 16; // Higher middle byte
252 data_f |= (uint32_t)data[2] << 8; // Lower middle byte
253 data_f |= data[3]; // Lowest byte
254
255 return data_f;
256}
