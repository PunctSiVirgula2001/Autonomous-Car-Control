15// Initialize PWM for SERVO
16void init_servo_pwm() {
17
18#if (MOTOR_CONTROL == LEDC)
19 // Prepare and set configuration of timers
20 ledc_timer_config_t ledc_timer = { .speed_mode = LEDC_Motor_And_Servo_MODE,
21 .duty_resolution = LEDC_SERVO_DUTY_RESOLUTION, .timer_num =
22 LEDC_SERVO_TIMER, .freq_hz = LEDC_Motor_And_Servo_FREQ, };
23 ledc_timer_config(&ledc_timer);
24
25 // Prepare individual configuration
26 ledc_channel_config_t ledc_channel = { .channel = LEDC_SERVO_CH0_CHANNEL,
27 .duty = 0, .gpio_num = LEDC_SERVO_GPIO, .speed_mode =
28 LEDC_Motor_And_Servo_MODE, .timer_sel = LEDC_SERVO_TIMER, };
29 ledc_channel_config(&ledc_channel);
30#elif (MOTOR_CONTROL == MCPWM)
31
32 // Configure MCPWM timer
33 mcpwm_timer_config_t config =
34 {
35 .group_id = 0,
36 .intr_priority = 0,
37 .clk_src = MCPWM_TIMER_CLK_SRC_DEFAULT,
38 // period_ticks=Desired Period * Resolution
39 // period_ticks= 20000 microseconds * 1 microsecond
40 // 1MHZ => 1 tick = 1/1000000 = 1 microsecond each tick
41 .resolution_hz = 1000000,
42 // 20000 * 1 microsecond = 20000 microseconds => 0.02 seconds => 50 hz
43 .period_ticks = 20000,
44 .count_mode = MCPWM_TIMER_COUNT_MODE_UP,
45 .flags.update_period_on_empty = true, // Update period when timer counts to zero
46 .flags.update_period_on_sync = false, // Do not update period on sync signal
47 };
48
49 ESP_ERROR_CHECK(mcpwm_new_timer(&config, &timer_servo));
50
51 // Configure MCPWM operator
52 mcpwm_operator_config_t operator_config = {
53 .group_id = 0,
54 .intr_priority = 0,
55 .flags.update_gen_action_on_tez = 1, // Update generator action when timer counts to zero
56 .flags.update_gen_action_on_tep = 0, // Do not update generator action when timer counts to peak
57 .flags.update_gen_action_on_sync = 0 // Do not update generator action on sync signal
58 };
59
60 ESP_ERROR_CHECK(mcpwm_new_operator(&operator_config, &operator_servo));
61 ESP_ERROR_CHECK(mcpwm_operator_connect_timer(operator_servo, timer_servo));
62
63 // Configure MCPWM comparator
64 mcpwm_comparator_config_t comparator_config =
65 {
66 .intr_priority = 0,
67 //This ensures that the compare value (duty cycle) is updated when the timer count reaches zero.
68 .flags.update_cmp_on_tez = 1,
69 .flags.update_cmp_on_tep = 0, // Do not update compare value when timer count equals peak
70 .flags.update_cmp_on_sync = 0 // Do not update compare value on sync event
71 };
72
73 ESP_ERROR_CHECK(mcpwm_new_comparator(operator_servo, &comparator_config, &comparator_servo));
74
75 // Configure MCPWM generator
76 mcpwm_generator_config_t generator_config = {
77 .gen_gpio_num = LEDC_SERVO_GPIO, // Set the GPIO number used by the generator
78 .flags.invert_pwm = false, // Do not invert the PWM signal
79 .flags.io_loop_back = false, // Disable loop-back mode
80 .flags.io_od_mode = false, // Do not configure as open-drain output
81 .flags.pull_up = false, // Disable internal pull-up resistor
82 .flags.pull_down = false, // Disable internal pull-down resistor
83 };
84
85 ESP_ERROR_CHECK(mcpwm_new_generator(operator_servo, &generator_config, &generator_servo));
86
87 // Set the generator actions
88 // go high on counter empty
89 ESP_ERROR_CHECK(mcpwm_generator_set_action_on_timer_event(generator_servo,
90 MCPWM_GEN_TIMER_EVENT_ACTION(MCPWM_TIMER_DIRECTION_UP, MCPWM_TIMER_EVENT_EMPTY, MCPWM_GEN_ACTION_HIGH)));
91 // go low on compare threshold
92 ESP_ERROR_CHECK(mcpwm_generator_set_action_on_compare_event(generator_servo,
93 MCPWM_GEN_COMPARE_EVENT_ACTION(MCPWM_TIMER_DIRECTION_UP, comparator_servo, MCPWM_GEN_ACTION_LOW)));
94
95 // Set the duty cycle for the servo
96 ESP_ERROR_CHECK(mcpwm_comparator_set_compare_value(comparator_servo, 1500)); // 1.5ms pulse width for neutral position
97
98 // Start the MCPWM timer
99 ESP_ERROR_CHECK(mcpwm_timer_enable(timer_servo));
100 ESP_ERROR_CHECK(mcpwm_timer_start_stop(timer_servo, MCPWM_TIMER_START_NO_STOP));
101
102#endif
103}

105// Update the PWM duty cycle for SERVO
106void update_servo_pwm(int pulse_width_us) {
107#if (MOTOR_CONTROL == LEDC)
108 uint32_t duty = (pulse_width_us * ((1 << LEDC_SERVO_DUTY_RESOLUTION) - 1)) / 20000;
109 ledc_set_duty(LEDC_Motor_And_Servo_MODE, LEDC_SERVO_CH0_CHANNEL, duty);
110 ledc_update_duty(LEDC_Motor_And_Servo_MODE, LEDC_SERVO_CH0_CHANNEL);
111#elif (MOTOR_CONTROL == MCPWM)
112 esp_err_t err = mcpwm_comparator_set_compare_value(comparator_servo, pulse_width_us);
113
114#endif
115}

117void changeSTEER(int value) {
