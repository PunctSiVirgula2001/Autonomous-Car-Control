75const float primarySmoothingFactor = 0.1; // Primary EMA smoothing factor
76const float secondarySmoothingFactor = 0.05; // Secondary EMA smoothing factor
77const float maxChangePerCycle = 5.0; // Maximum rate of output change per cycle

81// Compute PID with enhanced smoothing mechanisms
82void PID_Compute(PID_t *pid) {
83 float error = pid->setpoint - pid->measured;
84
85 // Proportional term
86 pid->P_term = pid->Kp * error;
87
88 // Integral term
89 pid->I_term += (pid->Ki * error);
90 clamp_float(&(pid->I_term), -40, 40); // Ensure I_term is within bounds
91
92#if low_pass_filter_derivative_chan == ON
93 // Derivative term with low-pass filter
94 float raw_D_term = error - pid->previous_error;
95 pid->D_term = pid->alpha * pid->previous_D_term + (1 - pid->alpha) * raw_D_term;
96 pid->D_term *= pid->Kd;
97#else
98 // Derivative term without low-pass filter
99 pid->D_term = pid->Kd * (error - pid->previous_error);
100#endif
101
102 // Calculate the proposed new output
103 float newOutput = pid->P_term + pid->I_term + pid->D_term;
104
105 // Apply primary Exponential Moving Average (EMA)
106 float primaryEMAOutput = primarySmoothingFactor * newOutput + (1 - primarySmoothingFactor) * pid->previousOutput;
107
108 // Apply secondary EMA for further smoothing
109 secondaryEMA = secondarySmoothingFactor * primaryEMAOutput + (1 - secondarySmoothingFactor) * secondaryEMA;
110
111 // Apply rate limiting to smooth large jumps
112 float outputDifference = secondaryEMA - pid->previousOutput;
113 if (fabs(outputDifference) > maxChangePerCycle) {
114 if (outputDifference > 0) {
115 pid->Output = pid->previousOutput + maxChangePerCycle;
116 } else {
117 pid->Output = pid->previousOutput - maxChangePerCycle;
118 }
119 } else {
120 pid->Output = secondaryEMA;
121 }
122
123 // Ensure the output is within the allowed range
124 clamp_int(&(pid->Output), -100, 100); // Ensure Output is within bounds
125
126 // Update previous values for next iteration
127 pid->previousOutput = pid->Output;
128 pid->previous_error = error;
129#if low_pass_filter_derivative_chan == ON
130 pid->previous_D_term = raw_D_term;
131#endif
132}

154/* Speed scaler for determining the correct threshold for distance of stop. */
155float get_speed_distance_sens_scaling(float speed, float speed_min, float speed_max) {
156 if (speed <= speed_min) {
157 return 1.0;
158 } else if (speed >= speed_max) {
159 return 5.0;
160 } else {
161 // Linear scaling between 1.0 and 5.0 for speeds between speed_min and speed_max
162 float t = (speed - speed_min) / (speed_max - speed_min); // Normalized speed in range [0, 1]
163 return 1.0 + t * (5.0 - 1.0); // Linear interpolation
164 }
165}

169/*SLIDING MEAN AVERAGE function*/
170double buffer[WINDOW_SIZE_SMA] = { 0.0 };
171double sum = 0.0;
172
173// Function to add value to the buffer and update the sliding mean
174double Sliding_Mean_Average(int newValue) {
175 // To keep track of the oldest value's index in the circular buffer
176 static int index = 0;
177 // Subtract the oldest value from the sum
178 sum -= buffer[index];
179 // Add the new value to the buffer and to the sum
180 buffer[index] = (double) newValue;
181 sum += newValue;
182 // Calculate the new mean
183 double mean = sum / (double) WINDOW_SIZE_SMA;
184 // Move the index to the next position (circularly)
185 index = (index + 1) % WINDOW_SIZE_SMA;
186 return mean;
187}
