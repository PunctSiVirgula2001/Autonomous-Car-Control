17void I2C_temperatureSensor_trigger_measurement(){
18 uint8_t calib[24];
19 I2C_transmit(I2C_temp_sens_dev_handle, (unsigned char[]){0x88},1);
20 vTaskDelay(pdMS_TO_TICKS(10));
21 I2C_receive(I2C_temp_sens_dev_handle, calib,24);
22 vTaskDelay(pdMS_TO_TICKS(10));
23 // Convert the bytes into calibration data
24 temp_calibration.dig_T1 = (uint16_t)(calib[1] << 8) | calib[0];
25 temp_calibration.dig_T2 = (int16_t)(calib[3] << 8) | calib[2];
26 temp_calibration.dig_T3 = (int16_t)(calib[5] << 8) | calib[4];
27
28 // Define the control register address for BMP280/BME280
29 uint8_t ctrl_meas_reg = 0xF4;
30 // Write a value to trigger a measurement: use 0x2X for force mode, 0x3X for normal mode
31 // The 'X' should be replaced with the combination of temperature and pressure oversampling
32 // Example: 0x27 for normal mode, oversampling x1 for both temperature and pressure
33 uint8_t ctrl_meas_value = 0x27;
34 // Transmit the address and the value
35 I2C_transmit(I2C_temp_sens_dev_handle, (unsigned char[]){ctrl_meas_reg, ctrl_meas_value},2);
36}

38void I2C_read_temperature(double *fine_temp)
39{
40 // Read temperature registers
41 uint8_t temp_msb;
42 uint8_t temp_lsb;
43 uint8_t temp_xlsb;
44 uint32_t raw_temp;
45 I2C_transmit(I2C_temp_sens_dev_handle, (unsigned char[]){0xFA},1);
46 I2C_receive(I2C_temp_sens_dev_handle, &temp_msb,8);
47
48 I2C_transmit(I2C_temp_sens_dev_handle, (unsigned char[]){0xFB},1);
49 I2C_receive(I2C_temp_sens_dev_handle, &temp_lsb,8);
50
51 I2C_transmit(I2C_temp_sens_dev_handle, (unsigned char[]){0xFC},1);
52 I2C_receive(I2C_temp_sens_dev_handle, &temp_xlsb,4);
53 // Combine bytes to get the raw temperature value
54 raw_temp = ((uint32_t)temp_msb << 12) | ((uint32_t)temp_lsb << 4) | (temp_xlsb >> 4);
55
56 double var1, var2;
57 var1 = (((double)raw_temp)/16384.0-((double)temp_calibration.dig_T1)/1024.0) * ((double)temp_calibration.dig_T2);
58 var2 = ((((double)raw_temp)/131072.0-((double)temp_calibration.dig_T1)/8192.0) *(((double)raw_temp)/131072.0-((double) temp_calibration.dig_T1)/8192.0)) * ((double)temp_calibration.dig_T3);
59 *fine_temp = (var1 + var2) / 5120.0;
60}
